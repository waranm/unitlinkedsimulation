สูตรเก่า vs สูตรใหม่
✗ เดิม: CAGR = (final / totalPremium)^(1/years) − 1
✓ ใหม่: หาค่า r ที่ทำให้ Σ premium × (1+r)^(T−t) = final
          IRR_annual = (1 + r_monthly)^12 − 1
Implementation
// Binary search — converges in 100 iterations (< 1 µs)
function calcIRR(premium, step, months, finalValue) {
  function fv(r) {
    let sum = 0;
    for (let m = 0; m < months; m += step)
      sum += premium * Math.pow(1 + r, months - m);
    return sum;
  }
  // fv() monotonically increases with r → binary search valid
  let lo = -0.9, hi = 0.3;
  for (let i = 0; i < 100; i++) {
    const mid = (lo + hi) / 2;
    if (fv(mid) < finalValue) lo = mid; else hi = mid;
  }
  return (Math.pow(1 + (lo + hi) / 2, 12) - 1) * 100;
}

// 1. Align monthly returns (intersection of all fund months)
const { fundOrder, returns } = getAlignedMonthlyReturns(navData);

// 2. Sample covariance matrix  Σ[i][j] = cov(r_i, r_j)
const covMatrix = computeCovMatrix(returns);   // divides by T−1

// 3. Cholesky: find L such that  L × Lᵀ = Σ
const L = cholesky(covMatrix);

// 4. Each month in runScenario():
const z = Array.from({ length: nFunds }, randNormal); // independent N(0,1)
for (let i = 0; i < nFunds; i++) {
  let shock = 0;
  for (let k = 0; k <= i; k++) shock += L[i][k] * z[k]; // w = L·z
  nav[funds[i]] *= Math.exp(mean_i + shock);              // shock ~ N(0, σ_i²)
}

Statistics Minor Fix
ปัญหา

ข้อมูล 47 เดือนที่ใช้คือ ตัวอย่าง จากประชากรจริงที่ไม่รู้จัก การหาร n (population formula) ให้ variance ที่ต่ำกว่าจริง (biased downward) เพราะ sample mean ถูก "pull" เข้าหาข้อมูล การหาร n−1 แก้ bias นี้ (Bessel's correction)

การเปลี่ยนแปลง
// ก่อน — population variance (biased)
const variance = n > 0
  ? returns.reduce((a,b) => a + (b-mean)**2, 0) / n
  : 0;

// หลัง — sample variance (unbiased)
const variance = n > 1
  ? returns.reduce((a,b) => a + (b-mean)**2, 0) / (n - 1)
  : 0;
ผลกระทบต่อแต่ละกองทุน (n = 47 เดือน)

Suite 8: Volatility Drag — Itô Correction

ตรวจสอบ property ทางคณิตศาสตร์ที่สำคัญ: เมื่อ μ_GBM = 0 (zero arithmetic drift, "fair game") และ σ > 0 median portfolio return ต้องติดลบ

GBM: dS/S = μ dt + σ dW, μ = 0
โดย Itô's lemma: log-return mean = μ − σ²/2 = −σ²/2 < 0
→ median NAV_t = NAV_0 · exp(−σ²t/2) < NAV_0 (ลดลงตามเวลา)
→ distribution of final portfolio value: right-skewed
→ arithmetic mean ≈ total paid (zero drift), แต่ P50 < total paid

Dynamic Mean vs Median Insight — ข้อความปรับตาม Gap อัตโนมัติ
js/app.js · renderOutcomeSummary() · mmcInsight()
Feature
เหตุผล

ข้อความเดิมเป็นข้อความคงที่ที่ไม่สะท้อนความรุนแรงของ skew จริงในผลลัพธ์ Portfolio ที่ mean ≈ median (กองทุนผันผวนน้อย) ควรได้รับข้อความที่ต่างจาก portfolio ที่ mean สูงกว่า median มากกว่า 20% (กองทุน equity ผันผวนสูง)

Logic
relGapPct = (mean − median) / median × 100

< 5% → Soft: "ค่าเฉลี่ยอาจสูงกว่าเล็กน้อย แต่ผลลัพธ์ส่วนใหญ่จะใกล้เคียงผลลัพธ์ทั่วไป"
5–20% → Balanced: "ค่าเฉลี่ยสูงขึ้นจากบางกรณีที่ได้ผลดีมาก คนส่วนใหญ่จะได้ผลลัพธ์ใกล้เคียงผลลัพธ์ทั่วไป"
> 20% → Strong: "ค่าเฉลี่ยไม่ใช่สิ่งที่คนส่วนใหญ่จะได้ ผลลัพธ์ทั่วไปสะท้อนความเป็นจริงได้ดีกว่า" 
function mmcInsight(relGap) {
  if (relGap === null) return '';
  if (relGap < 5)
    return 'ค่าเฉลี่ยอาจสูงกว่าเล็กน้อย แต่ผลลัพธ์ส่วนใหญ่จะใกล้เคียง'
         + '<strong>ผลลัพธ์ทั่วไป</strong> — ใช้ตัวเลขนี้เป็นหลักในการตัดสินใจ';
  if (relGap < 20)
    return 'ค่าเฉลี่ยสูงขึ้นจากบางกรณีที่ได้ผลดีมาก แต่คนส่วนใหญ่จะได้ผลลัพธ์ใกล้เคียง'
         + '<strong>ผลลัพธ์ทั่วไป</strong> — ควรใช้ตัวเลขนี้เป็นเกณฑ์หลัก';
  return 'ค่าเฉลี่ยไม่ใช่สิ่งที่คนส่วนใหญ่จะได้ ผลลัพธ์จริงมักต่ำกว่าค่าเฉลี่ยเพราะความผันผวน'
       + ' — <strong>ผลลัพธ์ทั่วไป</strong>สะท้อนความเป็นจริงได้ดีกว่า';
}
